Тема содержит переменные цветов и отступов а также набор иконок. Подробнее о [параметрах темы](https://www.ag-grid.com/react-data-grid/theming-parameters/).

```tsx
import { theme } from '@consta/ag-grid-adapter/theme';

import {
  AllCommunityModule,
  ClientSideRowModelModule,
  ModuleRegistry,
} from 'ag-grid-community';

import { AgGridReact } from 'ag-grid-react';

const ThemeMainExample = () => {
  const [rowData, setRowData] = useState<Item[]>([]);

  useEffect(() => {
    fetch('https://www.ag-grid.com/example-assets/olympic-winners.json')
      .then((resp) => resp.json())
      .then((data) => setRowData(data));
  }, []);

  return <AgGridReact {...otherOptions} rowData={rowData} theme={theme} />;
};
```

Вы можете изменить тему в случае необходимости. К примеру изменим фон не четной строки:

<MdxTabs>

```tsx
import { theme } from '@consta/ag-grid-adapter/theme';

import {
  AllCommunityModule,
  ClientSideRowModelModule,
  ModuleRegistry,
} from 'ag-grid-community';

import { AgGridReact } from 'ag-grid-react';

const myTheme = theme.withParams({
  oddRowBackgroundColor: 'var(--color-bg-stripe)',
});

const ThemeMainExample = () => {
  const [rowData, setRowData] = useState<Item[]>([]);

  useEffect(() => {
    fetch('https://www.ag-grid.com/example-assets/olympic-winners.json')
      .then((resp) => resp.json())
      .then((data) => setRowData(data));
  }, []);

  return <AgGridReact {...otherOptions} rowData={rowData} theme={myTheme} />;
};
```
